#include <iostream>
#include <cmath>
using namespace std;
int main(){
	int n;
	cin >> n;
	int arr[100][100];
	for (int i = 0; i < n; i++){
		for (int j = 0; j < n; j++){
			arr[i][j] = 0;
		}
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (i == 0) {
				arr[i][j] = 1;
			}
			else if (j == n - 1) {
				arr[i][j] = 1;
			}
			else if (i == n - 1) {
				arr[i][j] = 1;
			}
			else if (j == 0 && i != 1) {
				arr[i][j] = 1;
			}
			else if (i == 2 && j != n - 2) {
				arr[i][j] = 1;
			}
		}
	}
	int side = 1;
	int loop = pow(n, 2);
	while (loop > 0) {
		for (int i = 0; i < n; i++){
			for (int j = 0; j < n; j++){
				switch (side){
				case 1: {
					if ((((arr[i - 1][j + 1] == 0 && arr[i - 1][j] == 1) && arr[i + 1][j] != 1) && arr[i][j + 1] == 0) && arr[i][j + 2] == 1) {
						arr[i][j] = 1;
					}
					break;
				}
				case 2: {
					if ((((arr[i + 1][j + 1] == 0 && arr[i][j + 1] == 1) && arr[i][j - 1] != 1) && arr[i + 1][j] == 0) && arr[i + 2][j] == 1) {
						arr[i][j] = 1;
					}
					break;
				}
				case 3: {
					if ((((arr[i + 1][j - 1] == 0 && arr[i + 1][j] == 1) && arr[i - 1][j] != 1) && arr[i][j - 1] == 0) && arr[i][j - 2] == 1) {
						arr[i][j] = 1;
					}
					break;
				}
				case 4: {
					if ((((arr[i - 1][j - 1] == 0 && arr[i][j - 1] == 1) && arr[i][j + 1] != 1) && arr[i - 1][j] == 0) && arr[i - 2][j] == 1) {
						arr[i][j] = 1;
					}
					break;
				}
				}
			}
		}
		loop--;
		side++;
		if (side > 4) {
			side = 1;
		}
	}
	for (int i = 0; i < n; i++){
		for (int j = 0; j < n; j++){
			cout << arr[i][j] << ' ';
		}
		cout << endl;
	}
	return 0;
}
